webpack系列-1 webpack究竟是什么

发表于 2019-08-28

1 webpack究竟是什么？

以前写html,js的写法如下：

index.html文件

<!DOCTYPE html>
 <html lang="en">
 <head>
     <meta charset="UTF-8">
     <title>这是最原始的网页开发</title>
 </head>
 <body>
     <p>这是我们的网页内容</p>
     <div id='root'></div>
     <script src='./index.js'></script>
 </body>
 </html>

index.js文件

var dom = document.getElementById('root');
var header = document.createElement('div');
header.innerText = 'header';
dom.append(header);

var content = document.createElement('div');
content.innerText = 'content';
dom.append(content);

var sidebar = document.createElement('div');
sidebar.innerText = 'sidebar';
dom.append(sidebar);

随着业务增加，如果用这种面向过程的编程方式代码就会很长，所有的js逻辑都堆在一个文件，最后难以维护。
采用面向对象的编程方式

<!DOCTYPE html>
 <html lang="en">
 <head>
     <meta charset="UTF-8">
     <title>这是最原始的网页开发</title>
 </head>
 <body>
     <p>这是我们的网页内容</p>
     <div id='root'></div>
     <script src='./index.js'></script>
 </body>
 </html>

content.js文件

function Content() {
    var dom = document.getElementById('root');
    var content = document.createElement('div');
    content.innerText = 'content';
    dom.append(content);
}

module.exports = Content;

header.js文件

    function Header() {
    var dom = document.getElementById('root');
    var header = document.createElement('div');
    header.innerText = 'header';
    dom.append(header);
}

module.exports = Header;

sidebar.js

function Sidebar() {
    var dom = document.getElementById('root');
    var sidebar = document.createElement('div');
    sidebar.innerText = 'sidebar';
    dom.append(sidebar);
}

module.exports = Sidebar;

index.js

var Header = require('./header.js');
var Sidebar = require('./sidebar.js');
var Content = require('./content.js');

new Header();
new Sidebar();
new Content();

这个文件各司其职，便于维护管理。
不存在依赖颠倒出现的问题
运行上面代码
npm init

package.json文件 

"devDependencies": {
 "webpack-cli": "^3.1.2"
  },
 "dependencies": {
   "webpack": "^4.25.1"
 }

执行
npx webpack index.js
